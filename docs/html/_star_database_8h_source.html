88 static T ReadAtRecord(File& file, int16_t index) {
89 T record;
90 const size_t record_position = static_cast<size_t>(index) * sizeof(record);
91 if (!file.seek(record_position, SeekSet)) {
92 debugln(F("Failed to set the read position"));

94 }
95 if (file.read(reinterpret_cast<uint8_t*>(&record), sizeof(T)) !=
96 sizeof(T)) {
97 debugln(F("Failed to read record from file"));
98 return T(); // Return default value if read fails
99 }
